enable
configure terminal

! Hostname
hostname HQ-Router

! Inter-VLAN Routing
interface fa0/0.10
encapsulation dot1Q 10
ip address 192.168.146.1 255.255.255.0

interface fa0/0.20
encapsulation dot1Q 20
ip address 192.168.94.1 255.255.255.0

interface fa0/0.30
encapsulation dot1Q 30
ip address 192.168.129.1 255.255.255.0

interface fa0/0.40
encapsulation dot1Q 40
ip address 192.168.126.1 255.255.255.0

interface fa0/0.50
encapsulation dot1Q 50
ip address 192.168.1.1 255.255.255.0

! Voice VLAN
interface fa0/0.60
encapsulation dot1Q 60
ip address 192.168.60.1 255.255.255.0

! Internet Interface
interface g0/0/0
ip address 203.0.114.1 255.255.255.0
ip nat outside
no shutdown

! Inside Interface
interface fa0/0
ip nat inside

! DHCP
ip dhcp excluded-address 192.168.146.1 192.168.146.10
ip dhcp pool ADMIN
network 192.168.146.0 255.255.255.0
default-router 192.168.146.1

! NAT
access-list 1 permit 192.168.0.0 0.0.255.255
ip nat inside source list 1 interface g0/0/0 overload

! OSPF
router ospf 1
network 192.168.0.0 0.0.255.255 area 0

end
